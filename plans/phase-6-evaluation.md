### `rag/evaluation/ragas_runner.py`

**Purpose:** Run offline RAGAS evaluation using the local LLM as judge.
Does not send any data to the internet.

**RAGAS metrics used:**
- **Faithfulness:** Is every claim in the answer supported by the retrieved context?
- **Answer Relevancy:** Does the answer actually address the question?
- **Context Precision:** Are the retrieved passages relevant to the question?

RAGAS supports a custom LLM judge. We pass a wrapper around `LLMClient`.

```python
def run_evaluation(
    dataset: List[dict],
    config: RAGConfig,
    output_path: Optional[Path] = None,
    metrics: Optional[List[str]] = None,
) -> dict:
    """
    Run RAGAS evaluation on a dataset of question-answer pairs.

    Dataset format (from test_generator.py):
    [
        {
            "question": str,
            "ground_truth": str,       # The source chunk text
            "answer": str,             # Pipeline-generated answer
            "contexts": List[str],     # Retrieved passage texts used
            "source": str,
        },
        ...
    ]

    If "answer" and "contexts" are missing, they are generated by running
    the pipeline on each question first.

    Args:
        dataset:      List of eval dicts. Questions required; answers optional.
        config:       RAGConfig for LLM judge and pipeline.
        output_path:  Save JSON results to this path if provided.
        metrics:      Which RAGAS metrics to compute. Defaults to all three.

    Returns:
        Dict with keys: "faithfulness", "answer_relevancy", "context_precision",
        "n_evaluated", "timestamp"
    """
    from ragas import evaluate
    from ragas.metrics import faithfulness, answer_relevancy, context_precision
    from ragas.llms import LangchainLLMWrapper
    from datasets import Dataset as HFDataset

    metrics_map = {
        "faithfulness": faithfulness,
        "answer_relevancy": answer_relevancy,
        "context_precision": context_precision,
    }
    selected_metrics = [metrics_map[m] for m in (metrics or list(metrics_map.keys()))]

    # Step 1: Generate answers + contexts for items that don't have them
    pipeline = QueryPipeline(config)
    enriched = []
    for item in dataset:
        if "answer" not in item or "contexts" not in item:
            result = pipeline.answer(query=item["question"], history=[])
            item["answer"] = result.text
            item["contexts"] = [c.excerpt for c in result.citations]
        enriched.append(item)

    # Step 2: Build Hugging Face Dataset
    hf_data = HFDataset.from_list([
        {
            "question": e["question"],
            "answer": e["answer"],
            "contexts": e.get("contexts", [e.get("ground_truth", "")]),
            "ground_truth": e.get("ground_truth", ""),
        }
        for e in enriched
    ])

    # Step 3: Wrap local LLM as RAGAS judge
    llm_wrapper = _LocalLLMWrapper(config)

    # Step 4: Run RAGAS
    scores = evaluate(
        hf_data,
        metrics=selected_metrics,
        llm=llm_wrapper,
        raise_exceptions=False,
    )

    results = {
        "faithfulness": float(scores.get("faithfulness", 0.0)),
        "answer_relevancy": float(scores.get("answer_relevancy", 0.0)),
        "context_precision": float(scores.get("context_precision", 0.0)),
        "n_evaluated": len(enriched),
        "timestamp": datetime.utcnow().isoformat() + "Z",
    }

    if output_path:
        output_path.parent.mkdir(parents=True, exist_ok=True)
        with open(output_path, "w") as f:
            json.dump(results, f, indent=2)

    return results


class _LocalLLMWrapper:
    """
    Thin adapter that wraps LLMClient for use as a RAGAS LLM judge.
    RAGAS's LangchainLLMWrapper expects a .predict(text) -> str interface.
    """
    def __init__(self, config: RAGConfig) -> None:
        self._config = config

    def predict(self, text: str) -> str:
        llm = get_model_manager().get_llm(self._config)
        return llm.generate(text, max_tokens=200, temperature=0.1)

    async def apredict(self, text: str) -> str:
        return self.predict(text)  # sync fallback
```

### `rag/evaluation/latency_test.py`

**Purpose:** Measure P50 and P95 query latency over a set of test questions.

```python
def run_latency_test(
    questions: List[str],
    config: RAGConfig,
    warmup: int = 2,
) -> dict:
    """
    Measure query latency for a list of questions.

    Args:
        questions:  List of query strings.
        config:     RAGConfig.
        warmup:     Number of warmup queries to run before measurement
                    (models are cold on first call).

    Returns:
        {
            "p50_ms": float,
            "p95_ms": float,
            "p99_ms": float,
            "min_ms": float,
            "max_ms": float,
            "n_queries": int,
            "tier": str,
        }
    """
    pipeline = QueryPipeline(config)
    latencies = []

    all_questions = questions[:warmup] + questions  # prepend warmup queries
    for i, q in enumerate(all_questions):
        t0 = time.monotonic()
        pipeline.answer(query=q, history=[], show_sources=False)
        elapsed_ms = (time.monotonic() - t0) * 1000
        if i >= warmup:  # skip warmup
            latencies.append(elapsed_ms)

    latencies.sort()
    n = len(latencies)
    return {
        "p50_ms": latencies[int(n * 0.50)],
        "p95_ms": latencies[int(n * 0.95)],
        "p99_ms": latencies[int(n * 0.99)] if n >= 100 else latencies[-1],
        "min_ms": latencies[0],
        "max_ms": latencies[-1],
        "n_queries": n,
        "tier": config.resolved_tier,
    }
```

### `rag/storage/query_cache.py`

**Purpose:** Cache query answers to avoid re-running the full pipeline for
identical queries. Cache key = SHA-256 of (query + config hash).

**Privacy note:** The cache stores query strings and answers on disk in plaintext.
The user is warned on startup if caching is enabled (see `rag/cli.py` welcome screen).

**Storage:** `~/.ragdb/query_cache.db` (SQLite, LRU via indexed_at + COUNT LIMIT)

```python
CACHE_MAX_ENTRIES = 500

class QueryCache:

    def __init__(self, config: RAGConfig) -> None:
        self._db_path = config.db_root / "query_cache.db"
        self._enabled = config.storage.query_cache_enabled
        if self._enabled:
            self._init_db()

    def _init_db(self) -> None:
        self._db_path.parent.mkdir(parents=True, exist_ok=True)
        conn = sqlite3.connect(str(self._db_path))
        conn.execute("""
            CREATE TABLE IF NOT EXISTS cache (
                cache_key  TEXT PRIMARY KEY,
                query      TEXT NOT NULL,
                answer     TEXT NOT NULL,
                citations  TEXT NOT NULL,   -- JSON
                indexed_at TEXT NOT NULL
            )
        """)
        conn.commit()
        conn.close()

    def _cache_key(self, query: str) -> str:
        return hashlib.sha256(query.lower().strip().encode()).hexdigest()

    def get(self, query: str) -> Optional[AnswerResult]:
        """Return cached AnswerResult if present, else None."""
        if not self._enabled:
            return None
        key = self._cache_key(query)
        conn = sqlite3.connect(str(self._db_path))
        row = conn.execute(
            "SELECT answer, citations FROM cache WHERE cache_key = ?", (key,)
        ).fetchone()
        conn.close()
        if not row:
            return None
        answer_text, citations_json = row
        citations = [Citation(**c) for c in json.loads(citations_json)]
        return AnswerResult(
            text=answer_text,
            citations=citations,
            passages_used=len(citations),
        )

    def put(self, query: str, result: AnswerResult) -> None:
        """Store an AnswerResult in the cache. Evict oldest if over limit."""
        if not self._enabled:
            return
        key = self._cache_key(query)
        citations_json = json.dumps([
            {k: v for k, v in c.__dict__.items()} for c in result.citations
        ])
        conn = sqlite3.connect(str(self._db_path))
        conn.execute(
            "INSERT OR REPLACE INTO cache (cache_key, query, answer, citations, indexed_at) "
            "VALUES (?, ?, ?, ?, ?)",
            (key, query, result.text, citations_json, datetime.utcnow().isoformat())
        )
        # LRU eviction: keep only newest CACHE_MAX_ENTRIES
        count = conn.execute("SELECT COUNT(*) FROM cache").fetchone()[0]
        if count > CACHE_MAX_ENTRIES:
            conn.execute(
                "DELETE FROM cache WHERE cache_key IN "
                "(SELECT cache_key FROM cache ORDER BY indexed_at ASC LIMIT ?)",
                (count - CACHE_MAX_ENTRIES,)
            )
        conn.commit()
        conn.close()

    def clear(self) -> None:
        """Clear all cached entries."""
        if self._db_path.exists():
            conn = sqlite3.connect(str(self._db_path))
            conn.execute("DELETE FROM cache")
            conn.commit()
            conn.close()
```

### Update `rag/retrieval/bm25_index.py` — tantivy backend

**Auto-switch rule:** When `bm25_index.count() > 100_000`, switch to tantivy.

```python
TANTIVY_THRESHOLD = 100_000

class BM25Index:
    """
    Adaptive BM25 backend:
    - rank_bm25 (in-memory, pickle) for < 100K chunks
    - tantivy (memory-mapped, Rust) for >= 100K chunks

    Backend selection is automatic based on corpus size.
    The public API (add, search, delete, save) is identical for both backends.
    """

    def __init__(self, config: RAGConfig) -> None:
        self._config = config
        self._index_path = config.db_root / "bm25"
        self._tantivy_path = config.db_root / "tantivy_index"
        self._backend: Literal["rank_bm25", "tantivy"] = "rank_bm25"
        # ... existing Phase 1 init code ...
        self._check_and_maybe_migrate()

    def _check_and_maybe_migrate(self) -> None:
        """Migrate from rank_bm25 to tantivy if corpus exceeds threshold."""
        if self.count() >= TANTIVY_THRESHOLD and self._backend == "rank_bm25":
            self._migrate_to_tantivy()

    def _migrate_to_tantivy(self) -> None:
        """
        Build a tantivy index from the current corpus.
        After migration, all subsequent operations use tantivy.
        The rank_bm25 pickle is kept as a backup but not used.
        """
        try:
            import tantivy
        except ImportError:
            # tantivy not installed — warn and stay on rank_bm25
            logging.warning(
                "Corpus > 100K chunks but tantivy not installed. "
                "Install with: pip install tantivy. Staying on rank_bm25."
            )
            return

        logging.info(f"Migrating BM25 backend to tantivy ({self.count()} chunks)...")
        self._tantivy_path.mkdir(parents=True, exist_ok=True)

        schema_builder = tantivy.SchemaBuilder()
        schema_builder.add_text_field("text", stored=True)
        schema_builder.add_text_field("chunk_id", stored=True)
        schema = schema_builder.build()

        index = tantivy.Index(schema, path=str(self._tantivy_path))
        writer = index.writer(heap_size=50_000_000)

        for tokens, chunk_id in zip(self._corpus_tokens, self._chunk_ids):
            writer.add_document(tantivy.Document(
                text=" ".join(tokens),
                chunk_id=chunk_id,
            ))
        writer.commit()
        index.reload()

        self._tantivy_index = index
        self._backend = "tantivy"
        logging.info("Migration to tantivy complete.")

    def search(self, query: str, top_k: int = 20) -> List[Tuple[str, float]]:
        if self._backend == "tantivy":
            return self._search_tantivy(query, top_k)
        return self._search_rank_bm25(query, top_k)  # existing Phase 1 code

    def _search_tantivy(self, query: str, top_k: int) -> List[Tuple[str, float]]:
        searcher = self._tantivy_index.searcher()
        query_obj = self._tantivy_index.parse_query(query, ["text"])
        results = searcher.search(query_obj, top_k).hits
        return [
            (searcher.doc(addr)["chunk_id"][0], float(score))
            for score, addr in results
        ]
```

```python
def test_cache_hit_on_identical_query(minimal_config, sample_md, monkeypatch):
    # Enable cache
    minimal_config.storage.query_cache_enabled = True
    ingest_path(sample_md, config=minimal_config, recursive=False, console=None)

    pipeline = QueryPipeline(minimal_config)
    call_count = 0

    original_answer = pipeline._run_retrieval
    def counting_retrieval(*a, **kw):
        nonlocal call_count; call_count += 1
        return original_answer(*a, **kw)
    monkeypatch.setattr(pipeline, "_run_retrieval", counting_retrieval)

    pipeline.answer("test question", history=[])
    pipeline.answer("test question", history=[])  # identical
    assert call_count == 1  # retrieval only ran once

def test_cache_miss_on_different_query(minimal_config):
    minimal_config.storage.query_cache_enabled = True
    cache = QueryCache(minimal_config)
    assert cache.get("never asked this before") is None

def test_cache_lru_eviction(minimal_config):
    minimal_config.storage.query_cache_enabled = True
    cache = QueryCache(minimal_config)
    # Insert CACHE_MAX_ENTRIES + 1 items
    for i in range(501):
        cache.put(f"query {i}", AnswerResult(text=f"answer {i}", citations=[], passages_used=0))
    conn = sqlite3.connect(str(cache._db_path))
    count = conn.execute("SELECT COUNT(*) FROM cache").fetchone()[0]
    conn.close()
    assert count <= 500

def test_cache_disabled_by_default(minimal_config):
    assert minimal_config.storage.query_cache_enabled is False
    cache = QueryCache(minimal_config)
    cache.put("q", AnswerResult(text="a", citations=[], passages_used=0))
    assert cache.get("q") is None  # disabled — nothing stored
```
